  6 /**
  7 *	Returns the distance between two waypoints.
  8 * @return {Promise} - Promise object that represents 
  9 * the distance between two waypoints.
 10 */
 11 
 12 let getDistance2Points= (orig,dest) => {
 13 	return new Promise ((resolve,reject) => {
 14 		googleDistance.get(
 15 		{origin: [orig.lat,orig.lon].toString(),
 16 		 destination: [dest.lat,dest.lon].toString()
 17 		},
 18 		(err,data) => {
 19 			if(err) reject(err);
 20 			resolve(data.distanceValue);
 21 		});
 22 	})
 23 }

 26 /**
 27 * Returns the distances between each points of another array.
 28 * @param {array}
 29 * @return {Promise} - Promise object that represents an array
 30 * of the distances corresponding to the filter.

 32 
 33 let getDistances = points => {
 34 	let distances= [];
 35 	return new Promise ((resolve,reject) => {
 36 		for (var i=0; i< points.length -1; i++){
 37 			getDistance2Points(points[i],points[i+1])
 38 			.then ( value => {
 39 				distances.push(value);
 40 				if(distances.length == points.length -1)					
 41 					resolve(distances);
 42 			}, error => {
 43 				reject(err);
 44 			});					
 45 		}
 46 	})
 47 }
